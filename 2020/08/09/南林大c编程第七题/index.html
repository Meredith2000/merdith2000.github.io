题目

**设计一个FifoTable类表示环形队列，它有四个私有属性，分别为：
int table[SIZE]; //存放队列数据的数组
int tail; //队列尾指针
int head; //队列头指针
int max,min; //队列元素的界，也就是如果加入队列的元素值大于max或者小于min，加入操作无法成功。
三个公有成员函数如下
FifoTable(int max,int min); //结构函数，设置对象的max和min值。
bool AddElement(int d); //向队列加入一个元素d，成功返回真，否则返回假。如果队列满了，则新的值无法加入
int GetElement(); //从队列头获取一个元素，并返回该元素值。如果队列为空，则返回NULLVALUE
常量NULLVALUE和SIZE值如下
const int NULLVALUE= 999999;
const int SIZE = 10;
该类的代码完成后，如下main()函数可以执行。
void main()
{
for(i = 0; i < 9; i++)
cout << ft.GetElement() << “,”;

FifoTable ft(20,-2);
int i;

for(i = -2; i <= 23; i++)
ft.AddElement(i);

cout << ft.GetElement() << endl;
}
执行的结果为：
-2,-1,0,1,2,3,4,5,6,999999**

分析

这道题主要考察的是对队列的理解，我认为要掌握以下的几点

  • 判断队列是否为空
  • 判断队列是否已满
  • 修正指针

判断队列是否为空：

if (head == tail){
// 说明队列为空
}

判断队列是否已满：

if (head == (tail+1)%SIZE){
// 说明队列已满
}

修正指针：

if (head == SIZE){
head = 0;
}

if (tail == SIZE){
tail = 0;
}

代码

#include

using namespace std;

const int NULLVALUE= 999999;
const int SIZE = 10;

class FifoTable{
private:
int table[SIZE]; //存放队列数据的数组
int tail; //队列尾指针
int head; //队列头指针
int max,min; //队列元素的界，也就是如果加入队列的元素值大于max或者小于min，加入操作无法成功。
public:
FifoTable(int max,int min); //结构函数，设置对象的max和min值。
bool AddElement(int d); //向队列加入一个元素d，成功返回真，否则返回假。如果队列满了，则新的值无法加入
int GetElement(); //从队列头获取一个元素，并返回该元素值。如果队列为空，则返回NULLVALUE

};

FifoTable::FifoTable(int max, int min) {
tail = 0;
head = 0;
this->max = max;
this->min = min;
}

bool FifoTable::AddElement(int d) {
// 输入的数据不符合要求
if (d > max d < min) {
return false;
}
// 队列已经满了
if (head == (tail+1)%SIZE) {
return false;
}
table[tail] = d;
tail++;
// 由于是环形队列，所以再tail越界的时候要进行修正
if (tail == SIZE) {
tail = 0;
}
return true;
}

int FifoTable::GetElement() {
int temp;
// 栈空
if (head == tail) {
return NULLVALUE;
}
// 用来暂时保存要返回的值
temp = table[head];
head++;
// 由于是环形队列，所以再head越界的时候要进行修正
if (head == SIZE) {
head = 0;
}
return temp;
}

int main() {
FifoTable ft(20,-2);
int i;

for(i = -2; i <= 23; i++)
    ft.AddElement(i);

for(i = 0; i < 9; i++)
    cout << ft.GetElement() << ",";

cout << ft.GetElement() << endl;
return 0;

}
