分析

这道题主要考察的继承方面的知识。主要是要掌握如何进行类之间的继承，并且子类能够调用父类的构造器。同时还要了解三角形的面积计算公式

实现公有继承，并且子类调用父类的构造器：

// 父类
class father(){
father(); // 父类构造器
}

// 子类，继承父类
class son(): public father{
son(); // 子类构造器
}

// 调用父类构造器的子类构造器
son::son(): father(){
// 里面申明子类构造器
}

代码

#include
#include

using namespace std;

class polygon
{
private:
int polygonnum; //多边形的边数
public:
virtual double Getarea()=0; //求面积
virtual double Getperimeter()=0; //求周长
polygon(int n)
{
polygonnum = n;
};
int Getpolygonnum() //获取多边形的边数
{
return polygonnum;
}
};

// 公有继承
class triangle: public polygon{
private:
double x;
double y;
double z;
public:
triangle(double x, double y, double z);
double Getarea();
double Getperimeter();
};

// 定义子类的构造函数，并调用父类的构造函数
triangle::triangle(double x, double y, double z):polygon(3) {
this->x = x;
this->y = y;
this->z = z;
}

double triangle::Getarea() {
return sqrt((x+y+z)*(x+y-z)*(x+z-y)*(y+z-x))/4; // 三角形的计算公式
}

double triangle::Getperimeter() {
return x+y+z;
}

// ==================================

class rectangle: public polygon{
private:
double x;
double y;
public:
rectangle(double x, double y);
double Getarea();
double Getperimeter();
};

// 定义子类的构造函数，并调用父类的构造函数
rectangle::rectangle(double x, double y):polygon(4) {
this->x = x;
this->y = y;
}

double rectangle::Getarea() {
return x*y;
}

double rectangle::Getperimeter() {
return 2*(x+y);
}

int main() {
polygon *ptri = new triangle(3,5,5);
cout << ptri->Getpolygonnum() <<”, “<< ptri->Getarea()<<”, “<Getperimeter()<<endl;
polygon *prec = new rectangle(3,4);
cout << prec->Getpolygonnum()<<”, “<< prec->Getarea() <<”, “<Getperimeter()<<endl;
return 0;
}
