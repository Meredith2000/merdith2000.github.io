题目

设计一个Complex类的结构，该类表示复数。并设计和实现该类相应的结构函数和add成员函数，该函数完成复数的加法。该类完成后，如下main()函数可以执行。
void main()
{
Complex c1(3, 5);
Complex c2 = 4;
c1.add(c2);
c1.show();
c2.show();
Complex c3(0,5);
c3.show();
}
执行的结果为
7+5i
4
5i

分析

这道题没啥难度，只要是考察构造函数的使用还有类内部方法的设计。

代码

#include

using namespace std;

class Complex{
private:
int real;
int image;
public:
// 构造器
Complex(int real);
Complex(int real, int image);
void add(Complex c);
void show();
};

// 构造器和方法的具体实现
Complex::Complex(int real) {
this->real = real;
this->image = 0;
}

Complex::Complex(int real, int image) {
this->real = real;
this->image = image;
}

void Complex::add(Complex c) {
this->real += c.real;
this->image += c.image;
}

void Complex::show() {
if (real == 0) {
cout << image << ‘i’ << endl;
} else if (image == 0) {
cout << real << endl;
} else{
cout << real << ‘+’ << image << ‘i’ << endl;
}
}

int main()
{
Complex c1(3, 5); // 使用第二个构造器
Complex c2 = 4; // 使用第一个构造器
c1.add(c2); // 传入Complex对象，调用add方法
c1.show(); // 空参，调用add方法
c2.show(); // 同上
Complex c3(0,5); // 调用第二个构造器
c3.show(); // 空参，调用add方法
return 0;
}
